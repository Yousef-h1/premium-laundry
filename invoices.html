async function startSearch() { const query = document.getElementById('searchInput').value.trim(); if (!query) return; document.getElementById('searchInput').style.backgroundColor = "#fff9db"; try { const response = await fetch(WEBHOOK_URL_MAIN, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ action: "search", searchQuery: query, invoiceId: query }) }); const data = await response.json(); let res = Array.isArray(data) ? data[0] : data; if (res && res.json) res = res.json; if (res && (res["الاسم"] || res["الهاتف"])) { // تنظيف وجلب البيانات الأساسية document.getElementById('custName').value = cleanData(res["الاسم"]); document.getElementById('custPhone').value = cleanData(res["الهاتف"]); // تحديث السعر النهائي في الواجهة if (res["الاجمالي"]) { document.getElementById('finalTotal').innerText = cleanData(res["الاجمالي"]); } // جلب وتنظيف الخدمات if (res["الخدمات"]) { const rawServices = cleanData(res["الخدمات"]); const servicesList = rawServices.split(/,|\n/); // تقسيم بالفاصلة أو السطر الجديد const billBody = document.getElementById('billBody'); billBody.innerHTML = `
فاتورة سابقة: ${res["رقم الفاتورة"]}
`; servicesList.forEach(srv => { if(srv.trim()) { billBody.innerHTML += `
${srv.trim()}
`; } }); } document.getElementById('searchInput').style.backgroundColor = "#e8f5e9"; document.getElementById('searchInput').value = ""; } else { alert("⚠️ لم يتم العثور على بيانات"); document.getElementById('searchInput').style.backgroundColor = ""; } } catch (err) { console.error("Search Error:", err); alert("⚠️ خطأ في الاتصال بالخادم"); } }
